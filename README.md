# planets!

Controls:
	Arrows or WASD - fly
	"+"/"-", "Pinch" - zoom in / zoom out
	"Space" - fire weapon
	"Q" - switch weapon
	"O" - switch orbit style
	"P" - pause on/off
	"Backspace" - show/hide planet edit panel 
	"1" (while on planet) - randomize planet, same style 
	"2" (while on planet) - randomize planet, random style 
	"3" (while on planet) - randomize/remove planet rings
	"4" - randomize planetary system star
	"5" - randomize world
	"6" - randomize background
	"7" - randomize spaceship
	"H" - switch spaceship/observe mode
	"["/"]" - world speed decrease/increase
